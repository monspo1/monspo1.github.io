            //..............................................................................
            // function to prepare and make the geocode call
            //..............................................................................
            public function doGeocode(eatingExpAC:ArrayCollection):void
            {
                //Remove the busy cursor
                //this.cursorManager.removeBusyCursor();

                //===> set the busy cursor
                this.cursorManager.setBusyCursor();
                
                ////===>remove all shapes from the map
                //this.myMap.removeShapes();
                //this.myMap2.removeShapes();
                
                ////===> make an array to hold the addresses
                var arrAddresses:Array = new Array(); 
                
                
                for(var i:uint = 0; i < eatingExpAC.source.length; i++)
                {
                    arrAddresses.push(eatingExpAC[i].restLocation.toString());
                    trace("arrAddress["+i+"]: " + arrAddresses[i]);
                }
                
                
                //===> call the geocoder object's geocode method, passing the array of addresses
                if(isCurrentDPAboutPreviousEatingExp == true){
                    this.geocoder1.geocode(arrAddresses);
                    this.isFirstLoad4PreviousEatingExpMap = false;
                } else{
                    this.geocoder2.geocode(arrAddresses);
                    this.isFirstLoad4WantToEatMap = false;
                }
                
                
            }
            
            
            /*-------------------------------------------
            function called on a successful geocode response
            -------------------------------------------*/
            protected function onGeocodeResponse(event:GeocoderEvent):void
            {
                ////===> use the default cursor
                this.cursorManager.removeBusyCursor();
                
                ////===> clean the Array List
                this.alResults = new ArrayList();
                
                ////===> loop through the results grabbing each location
                for each(var loc:GeocoderLocation in event.geocoderResponse.locations)
                {
                    var o:Object = new Object();
                    o.LatLng = loc.displayLatLng.lat + ", " + loc.displayLatLng.lng;
                    o.GeocodeQuality = loc.geocodeQuality;
                    o.GeocodeQualityCode = loc.geocodeQualityCode;
                    
                    this.alResults.addItem(o);

                }
                
                ////=== InfoContent for each Poi
                var i:int = 0;
                var tempImage:Image = new Image();
                
                for each (var eachPoi:Poi in event.geocoderResponse.pois)
                {
                    ++i;
                    eachPoi.rolloverAndInfoTitleText = i + ". @" + eachPoi.infoContent; 
                    var tempFlexInfoWindow:MyFlexInfoWindow = new MyFlexInfoWindow();
                    
                    tempFlexInfoWindow.paddingLeft = 20;
                    tempFlexInfoWindow.paddingTop = myMap.height + 130;  
                    tempFlexInfoWindow.width = tabResults.width + tempFlexInfoWindow.paddingLeft;
                    tempFlexInfoWindow.height = 225;
                    tempImage.source = myMenuHistoryDP[--i].menuImgSrc;
                    tempImage.width = 25;
                    tempImage.height = 25;
                    tempFlexInfoWindow.addElement(tempImage);
