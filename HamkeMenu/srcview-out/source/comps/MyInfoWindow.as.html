package samples.customs
{
    import com.mapquest.tilemap.DefaultInfoWindow;
    import com.mapquest.tilemap.TileMap;
    
    import flash.text.TextFormat;
    
    public class MyInfoWindow extends DefaultInfoWindow
    {
        public function MyInfoWindow(map:TileMap)
        {
            super(map);
            
            //set some inital properties
            this.backgroundColor = 0xAF0000
            this.backgroundAlpha = 1.0;
            this.setContentTitleBuffer(15);

            //set text properties for the content area
            var tf:TextFormat = new TextFormat();
            tf.font = "Helvetica";
            tf.size = 10;
            tf.bold = true;
            tf.color = 0xfff000;
            this.setContentTextFormat(tf);
        }
        
        override protected function draw():void {    
            //keep this if statement here
            if (dirtyPlacement) {
                dirtyPlacement = false;
                placeItems();
            }
            
            //clear the current graphics
            this.bkg.graphics.clear();
            this.shadow.graphics.clear();

            //draw shadow for new ellipse info window
            if (useShadow) {
                this.shadow.graphics.beginFill(0x000000, .2);
                shadow.graphics.drawEllipse(shadowOffsetX-70, shadowOffsetY-50, backgroundWidth + 95, backgroundHeight + 75);
                this.shadow.graphics.endFill();
            }
            //draw background for new ellipse info window
            this.bkg.graphics.beginFill(backgroundColor, backgroundAlpha);
            bkg.graphics.drawEllipse(borderSize-70,borderSize-50,backgroundWidth + 90,backgroundHeight + 70);
            this.bkg.graphics.endFill();

            
            //set text properties for the title area
            var tf:TextFormat = new TextFormat();
            tf.font = "Helvetica";
            tf.size = 14;
            tf.bold = true;
            this.titleField.textColor = 0xfff000;
            this.titleBackground.visible = false;
            this.titleField.alpha = 0.9;
            this.titleField.defaultTextFormat = tf;
        }
    }
}
